Pairwise summation

You are given an array A containing N elements. You are required to perform Q queries of the following types on the array:

  • 0 L R: Print out the value of the required summation ∑j=LR∑k=j+1R(Aj∗Ak).
  • 1 X V: Change the value of the number that is at the Xth position to V.

Note: Indexing is 1 based

Example

Consider N = 3, A = {1, 2, 3}, Q = 2, and Queries = [[1,1,7], [0,1,3]]. You must determine the answer to each query:

  • For the 1st query we have, type = 1, X = 1, and V = 7. The modified array A = [7, 2, 3].

  • For the 2nd query we have, type = 0, L = 1, and R = 3. Thus the answer is: 7*2 + 7*3 + 2*3 = 41.

  • The next line contains N space-separated integers of array A.

Output format

Print the answer for each query of type 0 L R in a separate line.

#include<bits/stdc++.h>
using namespace std;

const long long MOD = 1e9+7;
const long long MAXN = 1e6+3;

long long f[MAXN][5];

long long calc(int n, int chr) {
    if (n == 1) return 1;
    if (f[n][chr] != 0) return f[n][chr];
    long long res = 0;
    for (int nextChr = chr; nextChr < 5; nextChr++) {
        res = (res + calc(n-1, nextChr)) % MOD;
    }
    return f[n][chr] = res;
}

long long solve (int n) {
   // Write your code here
   long long res = 0;
   for (int i = 0; i < 5; ++i) {
       res = (res + calc(n, i))%MOD;
   }
   return res;
}

int main() {

    ios::sync_with_stdio(0);
    cin.tie(0);
    int T;
    cin >> T;
    for(int t_i = 0; t_i < T; t_i++)
    {
        int n;
        cin >> n;

        long long out_;
        out_ = solve(n);
        cout << out_;
        cout << "\n";
    }
}
